#include <msp430.h>

// Extern functions and variables identified by the calls and moves
extern void Display_Process(int line, int string_ptr); 
extern unsigned char display_changed;

void Switches_Process(void) {
    // --- Check Switch 1 (P4IN & BIT1) ---
    // #2 corresponds to BIT1 (0x02)
    if (!(P4IN & BIT1)) { 
        Display_Process(0, 0);  // "Fall 2023"
        Display_Process(11, 0); // Offset for next string
        Display_Process(22, 0); 
        Display_Process(33, 0); 
        
        display_changed = 1;    // MOV.B #1 to the flag address
    }

    // --- Check Switch 2 (P2IN & BIT3) ---
    // #8 corresponds to BIT3 (0x08)
    if (!(P2IN & BIT3)) { 
        Display_Process(0, 0);  // Logic repeats for SW2 strings
        Display_Process(11, 0); 
        Display_Process(22, 0); 
        Display_Process(33, 0); 
        
        display_changed = 1; 
    }
}
